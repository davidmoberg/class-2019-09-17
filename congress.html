knitr::opts_chunk$set(echo = TRUE)
library(fivethirtyeight)
library(tidyverse)
library(ggplot2)
library(knitr)
congress_age%>%mutate(year = format(termstart, "%Y"))%>%
  filter(party %in% c("R", "D"))%>%
  group_by(year, party) %>%
  summarize(avg = mean(age))%>%
  ggplot(aes(x = year, y = avg, group = party, color = party))+geom_line()+labs(x = "year", y = "average age", title = "Average Age of Members of Congress")
